In my jQuery code, I first get the button, then store the "#para" DOM, and sections vector. 

$("button")
    .button()
    .click(function (event) {
        var parasDom = $("#paras");
        var sections = parasDom.children("section");

Then I get the tab DOM and append the <ul> to the tabs <div>. 

        // get tabs DOM 
        var tabsDom = $("#tabs");
        // create ul DOM
        tabsDom.append("
    ");
        var ulDom = tabsDom.find("ul");

    Next, I will iterate over the sections array, to append the content from paragraphs to the tabs elements. 

             // iterate over the sections array
        $.each(sections, function (i, sec) {
            i++;  // index start from 1
            // get the title
            var title = $(sec).find("header").text();
            // get the paragraph
            var para = $(sec).find("p");
            // add title name to the tab name
            ulDom.append("
  • " + title + "
    • ");
            // add paragraph content to the tab content
            tabsDom.append("
    " + "
    ");
            thisDiv = $("#tabs-" + i);
            thisDiv.append(para);
            console.log(thisDiv.text());
        });

    The line below is to activate the tabs UI: 

                 $("#tabs").tabs();

    After transformed, I will hide the button as well as the original paragraphs. 

                // Hide button
            $(this).attr("style", "display:None");
            // Hide original paragraphs
            $("div#paras").attr("style", "display: None");
